Cross Power Spectrum without noise

I = zeros(300,300);
I(50:100,70:120) = 255;
J = zeros(300,300);
J(120:170,20:90) = 255;

figure;
imagesc(I);
title('Image I');
colormap gray;

figure;
imagesc(J);
title('Image J');
colormap gray;

crossPowerSpectrum(I,J,0);
